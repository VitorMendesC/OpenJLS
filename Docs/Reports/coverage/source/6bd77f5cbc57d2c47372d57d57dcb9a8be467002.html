   356:       variable Last  : std_logic ;
 
   357:       variable NewTransfer : std_logic := '1' ;
 
   358:       variable Burst : std_logic ;
 
   359:       variable DelayCycles : integer ; 
 
   360:       variable BurstLength : integer := 0 ; 
 
   361:     begin
 
   362:       -- Initialize
 
   363:       TValid  <= '0' ;
 
   364:       TID     <= (TID'range => 'X') ;
 
   365:       TDest   <= (TDest'range => 'X') ;
 
   366:       TUser   <= (TUser'range => 'X') ;
 
   367:       TData   <= (TData'range => 'X') ;
 
   368:       TStrb   <= (TStrb'range => 'X') ;
 
   369:       TKeep   <= (TKeep'range => 'X') ;
 
   370:       TLast   <= 'X' ;
 
   371:       wait for 0 ns ; -- Allow Cov models to initialize 
 
   372:       wait for 0 ns ; -- Allow Cov models to initialize 
 
   373:       -- BurstLength - once per BurstLength, use BurstDelay, otherwise use BeatDelay
 
   374:       AddBins (BurstCov.BurstLengthCov,  80, GenBin(3,11,1)) ;     -- 80% Small Burst Length
 
   375:       AddBins (BurstCov.BurstLengthCov,  20, GenBin(109,131,1)) ;  -- 20% Large Burst Length
 
   376:       -- BurstDelay - happens at BurstLength boundaries
 
   377:       AddBins (BurstCov.BurstDelayCov,   80, GenBin(2,8,1)) ;   -- 80% Small delay
 
   378:       AddBins (BurstCov.BurstDelayCov,   20, GenBin(108,156,1)) ; -- 20% Large delay
 
   379:       -- BeatDelay - happens between each transfer it not at a BurstLength boundary
 
   380:       -- These are all small
 
   381:       AddBins (BurstCov.BeatDelayCov,    85, GenBin(0)) ;       -- 85%  0 Delay
 
   382:       AddBins (BurstCov.BeatDelayCov,    10, GenBin(1)) ;       -- 10%  1 Delay
 
   383:       AddBins (BurstCov.BeatDelayCov,     5, GenBin(2)) ;       --  5%  2 Delay
 
   384:   
 
   385:       TransmitLoop : loop
 
   386:         -- Find Transaction
 
   387:         if IsEmpty(TransmitFifo) then
 
   388:            WaitForToggle(TransmitRequestCount) ;
 
   389:         end if ;
 
   390:   
 
   391:         -- Get Transaction
 
   392:         (Burst, Data, ID, Dest, User, Last) := Pop(TransmitFifo) ;
 
   393:   
 
   394:         -- Delay between consecutive signaling of Valid
 
   395:         if UseCoverageDelays then 
 
   396:           -- BurstCoverage Delays
 
   397:           DelayCycles := GetRandDelay(BurstCov) ; 
 
   398:           WaitForClock(Clk, DelayCycles) ;
 
   399:         else
 
   400:           -- Deprecated static settings
 
   401:           if NewTransfer or not Burst then
 
   402:             WaitForClock(Clk, ValidDelayCycles) ; -- delay cycles
 
   403:           else
 
   404:             WaitForClock(Clk, ValidBurstDelayCycles) ;  -- beat delays
 
   405:           end if ;
 
   406:           NewTransfer := Last or not Burst ;
 
   407:         end if ; 
 
   408:   
 
   409:         -- Calculate Strb. 1 when data else 0
 
   410:         -- If Strb is unused it may be null range
 
   411:         for i in Strb'range loop
 
   412:           if Data(i*8) = 'W' or Data(i*8) = 'U' then
 
   413:             Strb(i) := '0' ;
 
   414:           else
 
   415:             Strb(i) := '1' ;
 
   416:           end if ;
 
   417:         end loop ;
 
   418:   
 
   419:         -- Calculate Keep.  1 when data /= 'U' else 0
 
   420:         -- If Keep is unused it may be null range
 
   421:         for i in Keep'range loop
 
   422:           if Data(i*8) = 'U' then
 
   423:             Keep(i) := '0' ;
 
   424:           else
 
   425:             Keep(i) := '1' ;
 
   426:           end if ;
 
   427:         end loop ;
 
   428:   
 
   429:         -- Do Transaction
 
   430:         TID     <= ID   after tpd_Clk_tid ;
 
   431:         TDest   <= Dest after tpd_Clk_TDest ;
 
   432:         TUser   <= User after tpd_Clk_TUser ;
 
   433:         TData   <= to_x01(Data) after tpd_Clk_TData ;
 
   434:         TStrb   <= Strb after tpd_Clk_TStrb ;
 
   435:         TKeep   <= Keep after tpd_Clk_TKeep ;
 
   436:         TLast   <= Last after tpd_Clk_TLast ;
 
   437:   
 
   438:         Log(ModelID,
 
   439:           "Axi Stream Send." &
 
   440:           "  TData: "     & to_hxstring(to_x01(Data)) &
 
   441:           ifelse(TStrb'length > 0, "  TStrb: "     & to_string( Strb), "") &
 
   442:           ifelse(TKeep'length > 0, "  TKeep: "     & to_string( Keep), "") &
 
   443:           ifelse(TID'length   > 0, "  TID: "       & to_hxstring(ID),   "") &
 
   444:           ifelse(TDest'length > 0, "  TDest: "     & to_hxstring(Dest), "") &
 
   445:           ifelse(TUser'length > 0, "  TUser: "     & to_hxstring(User), "") &
 
   446:           "  TLast: "     & to_string( Last) &
 
   447:           -- Must be DoneCount and not RequestCount due to queuing/Async and burst operations
 
   448:           "  Operation# " & to_string( TransmitDoneCount + 1),
 
   449:           INFO
 
   450:         ) ;
 
   451:   
 
   452:         ---------------------
 
   453:         DoAxiValidHandshake (
 
   454:         ---------------------
 
   455:           Clk            =>  Clk,
 
   456:           Valid          =>  TValid,
 
   457:           Ready          =>  TReady,
 
   458:           tpd_Clk_Valid  =>  tpd_Clk_TValid,
 
   459:           AlertLogID     =>  BusFailedID,
 
   460:           TimeOutMessage =>  "AXI Stream Send Operation # " & to_string(TransmitDoneCount + 1),
 
   461:           TimeOutPeriod  =>  TransmitReadyTimeOut * tperiod_Clk
 
   462:         ) ;
 
   463:   
 
   464:         -- State after transaction
 
   465:         TID     <= ID + 1    after tpd_Clk_tid ;
 
   466:         TDest   <= Dest + 1  after tpd_Clk_TDest ;
 
   467:         TUser   <= not User  after tpd_Clk_TUser ;
 
   468:         TData   <= not Data  after tpd_Clk_TData ;
 
   469:         TStrb   <= (TStrb'range => '1') after tpd_Clk_TStrb ;
 
   470:         TKeep   <= (TKeep'range => '1') after tpd_Clk_TKeep ;
 
   471:   
 
   472:         -- Signal completion
 
   473:         Increment(TransmitDoneCount) ;
 
   474:         wait for 0 ns ;
 
   475:       end loop TransmitLoop ;
 
   476:     end process TransmitHandler ;
 
   477:   end architecture SimpleTransmitter ;
