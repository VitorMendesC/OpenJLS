   419:         valid : std_logic; 
   420:         flush : std_logic 
   421:       ) is 
   422:       begin 
   423:    
   424:         iWord         <= w; 
   425:         iWordValidLen <= to_unsigned(len, VLEN_W); 
   426:         iWordValid    <= valid; 
   427:         iFlush        <= flush; 
   428:    
   429:         loop 
   430:    
   431:           wait until rising_edge(clk); 
   432:           exit when iStall = '0'; 
   433:    
   434:         end loop; 
   435:    
   436:         -- consumed on this edge; update reference model in DUT order (append then flush) 
   437:         if (valid = '1') then 
   438:           push_word(w, len); 
   439:         end if; 
   440:    
   441:         if (flush = '1') then 
   442:           finish_image; 
   443:         end if; 
   444:    
   445:         iWordValid <= '0'; 
   446:         iFlush     <= '0'; 
   447:    
   448:       end procedure send_beat; 
   449:    
   450:       -- One image: random beats; flush rides the LAST valid word (the DUT 
   451:       -- contract -- iFlush pulses on the cycle the final word is presented, never 
   452:       -- on an empty beat). Then wait for full drain before the next image. 
   453:    
   454:       procedure send_image ( 
   455:         beats : natural 
   456:       ) is 
   457:    
   458:         variable w   : std_logic_vector(IN_WIDTH - 1 downto 0); 
   459:         variable len : natural; 
   460:    
   461:       begin 
   462:    
   463:         for n in 1 to beats loop 
   464:    
   465:           -- ~30% all-ones words to manufacture 0xFF output bytes and stuff runs. 
   466:           if (rv.DistValInt(((1, 3), (0, 7))) = 1) then 
   467:             w := (others => '1'); 
   468:           else 
   469:             w := (others => '0'); 
   470:    
   471:             for k in 0 to IN_WIDTH / 8 - 1 loop 
   472:    
   473:               if (rv.DistValInt(((1, 1), (0, 7))) = 1) then 
   474:                 w(IN_WIDTH - 1 - k * 8 downto IN_WIDTH - (k + 1) * 8) := x"FF"; 
   475:               else 
   476:                 w(IN_WIDTH - 1 - k * 8 downto IN_WIDTH - (k + 1) * 8) := rv.RandSlv(0, 255, 8); 
   477:               end if; 
   478:    
   479:             end loop; 
   480:    
   481:           end if; 
   482:    
   483:           len := rv.RandInt(1, IN_WIDTH); 
   484:    
   485:           if (n = beats) then 
   486:             send_beat(w, len, '1', '1');   -- flush rides the final word 
   487:           else 
   488:             send_beat(w, len, '1', '0'); 
   489:           end if; 
   490:    
   491:         end loop; 
   492:    
   493:         sImagesSent <= sImagesSent + 1; 
   494:         -- serialize images: wait for full drain before the next one 
   495:         wait until sFlushDone = sImagesSent; 
   496:    
   497:       end procedure send_image; 
   498:    
   499:       -- Directed beat helper: a single word of `len` valid bits. 
   500:    
   501:       procedure directed ( 
   502:         w   : std_logic_vector(IN_WIDTH - 1 downto 0); 
   503:         len : natural 
   504:       ) is 
   505:       begin 
   506:    
   507:         send_beat(w, len, '1', '1'); 
   508:         sImagesSent <= sImagesSent + 1; 
   509:         wait until sFlushDone = sImagesSent; 
   510:    
   511:       end procedure directed; 
   512:    
   513:       variable ones : std_logic_vector(IN_WIDTH - 1 downto 0); 
   514:       variable wtmp : std_logic_vector(IN_WIDTH - 1 downto 0); 
   515:    
   516:     begin 
   517:    
   518:       -- Stim-driven inputs (no signal defaults; assign before reset). 
   519:       iWord         <= ZERO_W; 
   520:       iWordValid    <= '0'; 
   521:       iWordValidLen <= (others => '0'); 
   522:       iFlush        <= '0'; 
   523:    
   524:       -- Reference FF-stuffer state. 
   525:       curByte    := (others => '0'); 
   526:       bitsInByte := 0; 
   527:       anyBits    := false; 
   528:       lastEmitFF := false; 
   529:       realSince  := false; 
   530:       stuffSeen  := false; 
   531:    
   532:       SetAlertLogName("tb_byte_stuffer_osvvm"); 
   533:       SetLogEnable(PASSED, FALSE); 
   534:       rv.InitSeed(rv'instance_name); 
   535:    
   536:       covEmit := NewID("emitBytes"); 
   537:       AddBins(covEmit, "emitBytes", GenBin(0, OUT_BYTES, OUT_BYTES + 1)); 
   538:       -- FT_EMPTY is unreachable (see directed-corner note) and excluded. 
   539:       covFlush := NewID("flushType"); 
   540:       AddBins(covFlush, "flushType", GenBin(FT_CLEAN, FT_DANGLING, 3)); 
   541:       -- Require seeing data output BOTH with and without almost-full backpressure. 
   542:       -- (The 0-byte terminal beat only fires after the FIFO has drained, so 
   543:       -- 0-byte x almostFull is unreachable and deliberately not a bin.) 
   544:       covCross := NewID("data x almostFull"); 
   545:       AddCross(covCross, 
   546:                         "data x almostFull", 
   547:                         GenBin(1, 1, 1),                                                    -- data beat (emitBytes > 0) 
   548:                         GenBin(0, 1, 2)                                                     -- oAlmostFull 
   549:                       ); 
   550:       -- Stuff event per emitted byte: at least one image with no stuff, one with 
   551:       -- its first stuff, and one with a subsequent (run) stuff. Makes the FF->'0' 
   552:       -- stuffing -- the module's whole purpose -- visible in its own coverage. 
   553:       covStuff := NewID("stuffEvent"); 
   554:       AddBins(covStuff, "noStuff",    GenBin(SE_NONE, SE_NONE)); 
   555:       AddBins(covStuff, "firstStuff", GenBin(SE_FIRST, SE_FIRST)); 
   556:       AddBins(covStuff, "runStuff",   GenBin(SE_RUN, SE_RUN)); 
   557:    
   558:       apply_reset(clk, rst, 6, '1'); 
   559:       ones := (others => '1'); 
   560:    
   561:       -------------------------------------------------------------------------- 
   562:       -- Directed corners (regression locks). 
   563:       -------------------------------------------------------------------------- 
   564:       -- NOTE: a truly-empty flush (zero payload bits) is intentionally NOT tested 
   565:       -- -- it is unreachable. Every valid word into byte_stuffer carries >=1 bit 
   566:       -- (regular/RI: the Golomb unary stop bit in the bit_packer; token_raw: 
   567:       -- A15_A16 guarantees raw suffix len >=1), and flush rides the last valid 
   568:       -- word, so the accumulator is never empty at flush. 
   569:    
   570:       -- Single non-FF byte, byte-aligned clean end. 
   571:       wtmp                                   := (others => '0'); 
   572:       wtmp(IN_WIDTH - 1 downto IN_WIDTH - 8) := x"5A"; 
   573:       directed(wtmp, 8); 
   574:    
   575:       -- Single 0xFF byte -> trailing stuffed 0x00 (dangling-FF terminal). 
   576:       wtmp                                   := (others => '0'); 
   577:       wtmp(IN_WIDTH - 1 downto IN_WIDTH - 8) := x"FF"; 
   578:       directed(wtmp, 8); 
   579:    
   580:       -- Sub-byte residue (5 real bits) padded post-stuffing. 
   581:       wtmp                                   := (others => '0'); 
   582:       wtmp(IN_WIDTH - 1 downto IN_WIDTH - 5) := "10101"; 
   583:       directed(wtmp, 5); 
   584:    
   585:       -- Long 0xFF run in one beat: forces consecutive stuffs across the 4-slot 
   586:       -- chain (full-width all-ones). 
   587:       directed(ones, IN_WIDTH); 
   588:    
   589:       -- Multi-word flush with residue: > FIFO width of all-ones, flushed on a 
   590:       -- final partial beat (regression for the multi-word valid-bits overflow). 
   591:       send_beat(ones, IN_WIDTH, '1', '0'); 
   592:       send_beat(ones, IN_WIDTH, '1', '0'); 
   593:       wtmp                                    := (others => '0'); 
   594:       wtmp(IN_WIDTH - 1 downto IN_WIDTH - 12) := x"FF" & "0101"; 
   595:       directed(wtmp, 12); 
   596:    
   597:       -- Back-to-back tiny images (context reset between flushes). 
   598:       for n in 1 to 4 loop 
   599:    
   600:         wtmp                                   := (others => '0'); 
   601:         wtmp(IN_WIDTH - 1 downto IN_WIDTH - 8) := std_logic_vector(to_unsigned(n * 32, 8)); 
   602:         directed(wtmp, 8); 
   603:    
   604:       end loop; 
   605:    
   606:       -------------------------------------------------------------------------- 
   607:       -- Constrained-random images. 
   608:       -------------------------------------------------------------------------- 
   609:       for img in 1 to N_RANDOM loop 
   610:    
   611:         send_image(rv.RandInt(1, MAX_BEATS)); 
   612:    
   613:       end loop; 
   614:    
   615:       -------------------------------------------------------------------------- 
   616:       -- Mid-operation iRst: drive a partial image (valid beats, no flush) then 
   617:       -- assert iRst before the flush. The aborted bits are never scored (sIgnore); 
   618:       -- after reset the byte_stuffer must be idle, and a fresh image must come out 
   619:       -- byte-exact. The reference FF-stuffer state is clean here (last image was 
   620:       -- flushed) and is re-initialised below to mirror the reset. 
   621:       -------------------------------------------------------------------------- 
   622:       sIgnore <= true; 
   623:    
   624:       for k in 1 to 4 loop 
   625:    
   626:         iWord         <= rv.RandSlv(IN_WIDTH); 
   627:         iWordValidLen <= to_unsigned(IN_WIDTH, VLEN_W); 
   628:         iWordValid    <= '1'; 
   629:         iFlush        <= '0'; 
   630:    
   631:         loop 
   632:    
   633:           wait until rising_edge(clk); 
   634:           exit when iStall = '0'; 
   635:    
   636:         end loop; 
   637:    
   638:       end loop; 
   639:    
   640:       iWordValid <= '0'; 
   641:       iFlush     <= '0'; 
   642:       apply_reset(clk, rst, 6, '1'); 
   643:       wait for 1 ns; 
   644:       AffirmIf(oWordValid = '0', "mid-op reset: byte_stuffer output cleared"); 
   645:       AffirmIf(oFlushDone = '0', "mid-op reset: no flush-done after reset"); 
   646:    
   647:       -- Re-initialise the reference FF-stuffer state to mirror the reset. 
   648:       curByte    := (others => '0'); 
   649:       bitsInByte := 0; 
   650:       anyBits    := false; 
   651:       lastEmitFF := false; 
   652:       realSince  := false; 
   653:       stuffSeen  := false; 
   654:    
   655:       sIgnore <= false; 
   656:       wait until rising_edge(clk); 
   657:    
   658:       -- Recovery: a fresh image must encode byte-exact through the scoreboard. 
   659:       send_image(rv.RandInt(1, MAX_BEATS)); 
   660:    
   661:       -------------------------------------------------------------------------- 
   662:       -- Done: every image already drained (each send_image/directed waits for its 
   663:       -- own oFlushDone), so just let the monitor settle, then report. 
   664:       -------------------------------------------------------------------------- 
   665:       sDriverDone <= true; 
   666:       wait for 10 * CLK_PERIOD; 
   667:    
   668:       AffirmIf(IsEmpty(SB_ID), "scoreboard drained (all expected bytes consumed)"); 
   669:       AffirmIfEqual(GetErrorCount(SB_ID), 0, "scoreboard mismatches"); 
   670:    
   671:       WriteBin(covEmit); 
   672:       WriteBin(covFlush); 
   673:       WriteBin(covCross); 
   674:       WriteBin(covStuff); 
   675:       AffirmIf(IsCovered(covEmit), "emitBytes coverage closed"); 
   676:       AffirmIf(IsCovered(covFlush), "flushType coverage closed"); 
   677:       AffirmIf(IsCovered(covCross), "data x almostFull coverage closed"); 
   678:       AffirmIf(IsCovered(covStuff), "stuff-event coverage closed"); 
   679:    
   680:       end_of_test("tb_byte_stuffer_osvvm");
