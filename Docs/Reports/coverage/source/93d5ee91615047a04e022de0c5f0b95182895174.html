   435:       procedure send_image ( 
   436:         beats : natural 
   437:       ) is 
   438:    
   439:         variable w   : std_logic_vector(IN_WIDTH - 1 downto 0); 
   440:         variable len : natural; 
   441:    
   442:       begin 
   443:    
   444:         for n in 1 to beats loop 
   445:    
   446:           -- ~30% all-ones words to manufacture 0xFF output bytes and stuff runs. 
   447:           if (rv.DistValInt(((1, 3), (0, 7))) = 1) then 
   448:             w := (others => '1'); 
   449:           else 
   450:             w := (others => '0'); 
   451:    
   452:             for k in 0 to IN_WIDTH / 8 - 1 loop 
   453:    
   454:               if (rv.DistValInt(((1, 1), (0, 7))) = 1) then 
   455:                 w(IN_WIDTH - 1 - k * 8 downto IN_WIDTH - (k + 1) * 8) := x"FF"; 
   456:               else 
   457:                 w(IN_WIDTH - 1 - k * 8 downto IN_WIDTH - (k + 1) * 8) := rv.RandSlv(0, 255, 8); 
   458:               end if; 
   459:    
   460:             end loop; 
   461:    
   462:           end if; 
   463:    
   464:           len := rv.RandInt(1, IN_WIDTH); 
   465:    
   466:           if (n = beats) then 
   467:             send_beat(w, len, '1', '1');   -- flush rides the final word 
   468:           else 
   469:             send_beat(w, len, '1', '0'); 
   470:           end if; 
   471:    
   472:         end loop; 
   473:    
   474:         sImagesSent <= sImagesSent + 1; 
   475:         -- serialize images: wait for full drain before the next one 
   476:         wait until sFlushDone = sImagesSent; 
   477:    
   478:       end procedure send_image; 
   479:    
   480:       -- Directed beat helper: a single word of `len` valid bits. 
   481:    
   482:       procedure directed ( 
   483:         w   : std_logic_vector(IN_WIDTH - 1 downto 0); 
   484:         len : natural 
   485:       ) is 
   486:       begin 
   487:    
   488:         send_beat(w, len, '1', '1'); 
   489:         sImagesSent <= sImagesSent + 1; 
   490:         wait until sFlushDone = sImagesSent; 
   491:    
   492:       end procedure directed; 
   493:    
   494:       variable ones : std_logic_vector(IN_WIDTH - 1 downto 0); 
   495:       variable wtmp : std_logic_vector(IN_WIDTH - 1 downto 0); 
   496:    
   497:     begin 
   498:    
   499:       -- Stim-driven inputs (no signal defaults; assign before reset). 
   500:       iWord         <= ZERO_W; 
   501:       iWordValid    <= '0'; 
   502:       iWordValidLen <= (others => '0'); 
   503:       iFlush        <= '0'; 
   504:    
   505:       -- Reference FF-stuffer state. 
   506:       curByte    := (others => '0'); 
   507:       bitsInByte := 0; 
   508:       anyBits    := false; 
   509:       lastEmitFF := false; 
   510:       realSince  := false; 
   511:    
   512:       SetAlertLogName("tb_byte_stuffer_osvvm"); 
   513:       SetLogEnable(PASSED, FALSE); 
   514:       rv.InitSeed(rv'instance_name); 
   515:    
   516:       covEmit := NewID("emitBytes"); 
   517:       AddBins(covEmit, "emitBytes", GenBin(0, OUT_BYTES, OUT_BYTES + 1)); 
   518:       -- FT_EMPTY is unreachable (see directed-corner note) and excluded. 
   519:       covFlush := NewID("flushType"); 
   520:       AddBins(covFlush, "flushType", GenBin(FT_CLEAN, FT_DANGLING, 3)); 
   521:       -- Require seeing data output BOTH with and without almost-full backpressure. 
   522:       -- (The 0-byte terminal beat only fires after the FIFO has drained, so 
   523:       -- 0-byte x almostFull is unreachable and deliberately not a bin.) 
   524:       covCross := NewID("data x almostFull"); 
   525:       AddCross(covCross, 
   526:                         "data x almostFull", 
   527:                         GenBin(1, 1, 1),                                                    -- data beat (emitBytes > 0) 
   528:                         GenBin(0, 1, 2)                                                     -- oAlmostFull 
   529:                       ); 
   530:    
   531:       apply_reset(clk, rst, 6, '1'); 
   532:       ones := (others => '1'); 
   533:    
   534:       -------------------------------------------------------------------------- 
   535:       -- Directed corners (regression locks). 
   536:       -------------------------------------------------------------------------- 
   537:       -- NOTE: a truly-empty flush (zero payload bits) is intentionally NOT tested 
   538:       -- -- it is unreachable. Every valid word into byte_stuffer carries >=1 bit 
   539:       -- (regular/RI: the Golomb unary stop bit in the bit_packer; token_raw: 
   540:       -- A15_A16 guarantees raw suffix len >=1), and flush rides the last valid 
   541:       -- word, so the accumulator is never empty at flush. 
   542:    
   543:       -- Single non-FF byte, byte-aligned clean end. 
   544:       wtmp                                   := (others => '0'); 
   545:       wtmp(IN_WIDTH - 1 downto IN_WIDTH - 8) := x"5A"; 
   546:       directed(wtmp, 8); 
   547:    
   548:       -- Single 0xFF byte -> trailing stuffed 0x00 (dangling-FF terminal). 
   549:       wtmp                                   := (others => '0'); 
   550:       wtmp(IN_WIDTH - 1 downto IN_WIDTH - 8) := x"FF"; 
   551:       directed(wtmp, 8); 
   552:    
   553:       -- Sub-byte residue (5 real bits) padded post-stuffing. 
   554:       wtmp                                   := (others => '0'); 
   555:       wtmp(IN_WIDTH - 1 downto IN_WIDTH - 5) := "10101"; 
   556:       directed(wtmp, 5); 
   557:    
   558:       -- Long 0xFF run in one beat: forces consecutive stuffs across the 4-slot 
   559:       -- chain (full-width all-ones). 
   560:       directed(ones, IN_WIDTH); 
   561:    
   562:       -- Multi-word flush with residue: > FIFO width of all-ones, flushed on a 
   563:       -- final partial beat (regression for the multi-word valid-bits overflow). 
   564:       send_beat(ones, IN_WIDTH, '1', '0'); 
   565:       send_beat(ones, IN_WIDTH, '1', '0'); 
   566:       wtmp                                    := (others => '0'); 
   567:       wtmp(IN_WIDTH - 1 downto IN_WIDTH - 12) := x"FF" & "0101"; 
   568:       directed(wtmp, 12); 
   569:    
   570:       -- Back-to-back tiny images (context reset between flushes). 
   571:       for n in 1 to 4 loop 
   572:    
   573:         wtmp                                   := (others => '0'); 
   574:         wtmp(IN_WIDTH - 1 downto IN_WIDTH - 8) := std_logic_vector(to_unsigned(n * 32, 8)); 
   575:         directed(wtmp, 8); 
   576:    
   577:       end loop; 
   578:    
   579:       -------------------------------------------------------------------------- 
   580:       -- Constrained-random images. 
   581:       -------------------------------------------------------------------------- 
   582:       for img in 1 to N_RANDOM loop 
   583:    
   584:         send_image(rv.RandInt(1, MAX_BEATS)); 
   585:    
   586:       end loop; 
   587:    
   588:       -------------------------------------------------------------------------- 
   589:       -- Mid-operation iRst: drive a partial image (valid beats, no flush) then 
   590:       -- assert iRst before the flush. The aborted bits are never scored (sIgnore); 
   591:       -- after reset the byte_stuffer must be idle, and a fresh image must come out 
   592:       -- byte-exact. The reference FF-stuffer state is clean here (last image was 
   593:       -- flushed) and is re-initialised below to mirror the reset. 
   594:       -------------------------------------------------------------------------- 
   595:       sIgnore <= true; 
   596:    
   597:       for k in 1 to 4 loop 
   598:    
   599:         iWord         <= rv.RandSlv(IN_WIDTH); 
   600:         iWordValidLen <= to_unsigned(IN_WIDTH, VLEN_W); 
   601:         iWordValid    <= '1'; 
   602:         iFlush        <= '0'; 
   603:    
   604:         loop 
   605:    
   606:           wait until rising_edge(clk); 
   607:           exit when iStall = '0'; 
   608:    
   609:         end loop; 
   610:    
   611:       end loop; 
   612:    
   613:       iWordValid <= '0'; 
   614:       iFlush     <= '0'; 
   615:       apply_reset(clk, rst, 6, '1'); 
   616:       wait for 1 ns; 
   617:       AffirmIf(oWordValid = '0', "mid-op reset: byte_stuffer output cleared"); 
   618:       AffirmIf(oFlushDone = '0', "mid-op reset: no flush-done after reset"); 
   619:    
   620:       -- Re-initialise the reference FF-stuffer state to mirror the reset. 
   621:       curByte    := (others => '0'); 
   622:       bitsInByte := 0; 
   623:       anyBits    := false; 
   624:       lastEmitFF := false; 
   625:       realSince  := false; 
   626:    
   627:       sIgnore <= false; 
   628:       wait until rising_edge(clk); 
   629:    
   630:       -- Recovery: a fresh image must encode byte-exact through the scoreboard. 
   631:       send_image(rv.RandInt(1, MAX_BEATS)); 
   632:    
   633:       -------------------------------------------------------------------------- 
   634:       -- Done: every image already drained (each send_image/directed waits for its 
   635:       -- own oFlushDone), so just let the monitor settle, then report. 
   636:       -------------------------------------------------------------------------- 
   637:       sDriverDone <= true; 
   638:       wait for 10 * CLK_PERIOD; 
   639:    
   640:       AffirmIf(IsEmpty(SB_ID), "scoreboard drained (all expected bytes consumed)"); 
   641:       AffirmIfEqual(GetErrorCount(SB_ID), 0, "scoreboard mismatches"); 
   642:    
   643:       WriteBin(covEmit); 
   644:       WriteBin(covFlush); 
   645:       WriteBin(covCross); 
   646:       AffirmIf(IsCovered(covEmit), "emitBytes coverage closed"); 
   647:       AffirmIf(IsCovered(covFlush), "flushType coverage closed"); 
   648:       AffirmIf(IsCovered(covCross), "data x almostFull coverage closed"); 
   649:    
   650:       end_of_test("tb_byte_stuffer_osvvm"); 
   651:       wait; 
   652:    
   653:     end process stim_proc;
